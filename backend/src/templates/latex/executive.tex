% ── Entry command (for education, achievements, etc) ──────────────────────────
\newcommand{\cventry}[4]{%
  % #1: Title/Position, #2: Organization/School, #3: Date, #4: Description
  \noindent\begin{minipage}[t]{0.75\linewidth}
    {\textbf{#1}}\\[1pt]
    {\footnotesize\color{mediumgray}#2}
    \ifx&#4&\else\\[-4pt]{\footnotesize\color{mediumgray}#4}\fi
  \end{minipage}%
  \begin{minipage}[t]{0.25\linewidth}
    \raggedleft{\footnotesize\color{lightblue}\textbf{#3}}
  \end{minipage}%
  \vspace{6pt}%
}
