% ── Section heading: gold dot + Cormorant-style serif title + rule ────────────
% Usage: \execsec{Title}
\newcommand{\execsec}[1]{%
  \vspace{10pt}%
  \noindent
  \begin{tikzpicture}[baseline=-2pt]
    \fill[gold] (0,0) circle (3.2pt);
  \end{tikzpicture}%
  \hspace{6pt}%
  {\fontsize{14pt}{16pt}\selectfont\bfseries\color{charcoal} #1}%
  \hspace{6pt}%
  \leaders\hrule height 0.6pt\hfill\kern0pt%
  \vspace{5pt}%
  \newline%
}

% ── Education entry ───────────────────────────────────────────────────────────
\newcommand{\eduentry}[4]{%
  % #1 degree, #2 institution+stream, #3 year, #4 pct
  \noindent\makebox[\linewidth]{%
    \begin{minipage}[t]{0.72\linewidth}
      {\small\bfseries\color{charcoal} #1}\\
      {\footnotesize\color{muted} #2}
    \end{minipage}%
    \hfill
    \begin{minipage}[t]{0.24\linewidth}
      \raggedleft
      {\small\bfseries\color{charcoal2} #3}\\
      {\footnotesize\bfseries\color{gold} #4}
    \end{minipage}%
  }%
  \vspace{4pt}%
  \noindent{\color{linecol}\rule{\linewidth}{0.5pt}}%
  \vspace{4pt}\\
}

% ── Dot-leader row (achievements, sports) ────────────────────────────────────
\newcommand{\leaderrow}[3]{%
  % #1 title, #2 sub, #3 date-or-level
  \noindent\makebox[\linewidth]{%
    {\small\bfseries\color{charcoal} #1}%
    \ifx&#2&\else{\small\color{muted} \space— #2}\fi%
    \leaders\hbox{\footnotesize\color{linecol}.}\hfill\kern0pt%
    {\small\bfseries\color{gold} #3}%
  }%
  \vspace{3pt}%
  \noindent{\color{linecol}\rule{\linewidth}{0.5pt}}%
  \vspace{3pt}\\
}

}
