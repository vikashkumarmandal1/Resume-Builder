% ── Left-bar section heading ──────────────────────────────────────────────────
% Usage: \classsec{Title}
\newcommand{\classsec}[1]{%
  \vspace{10pt}%
  \noindent
  \begin{tikzpicture}[baseline=(t.base)]
    \fill[emerald] (0,-0.05) rectangle (0.18, 0.72);
    \node[anchor=west, inner sep=0pt] (t) at (0.26,0.32)
      {\fontsize{8pt}{9pt}\selectfont\bfseries\color{slatemd}\MakeUppercase{#1}};
  \end{tikzpicture}%
  \vspace{4pt}%
  \newline%
}

% ── Education row: degree | institution | year | pct ─────────────────────────
\newcommand{\edurow}[4]{%
  \noindent\makebox[\linewidth]{%
    \begin{minipage}[t]{0.72\linewidth}
      {\small\bfseries\color{slate} #1}\\
      {\footnotesize\color{muted} #2}
    \end{minipage}%
    \hfill
    \begin{minipage}[t]{0.24\linewidth}
      \raggedleft
      {\small\bfseries\color{slatemd} #3}\\
      {\footnotesize\bfseries\color{emerald} #4}
    \end{minipage}%
  }%
  \vspace{4pt}%
  \noindent{\color{line}\rule{\linewidth}{0.4pt}}%
  \vspace{4pt}\\
}

}
